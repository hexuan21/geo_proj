class Queue(collections.deque):
    def __init__(self):
        super(Queue, self).__init__()
        self.__head = self.__tail = None

    def is_empty(self):
        return self.__head is None

    def is_full(self):
        return self.__tail is not None

    def enqueue(self, value):
        if self.__head is None:
            self.__head = self.__tail = value
        else:
            self.__tail.next = value
            self.__tail = value

    def dequeue(self):
        value = self.__head
        if value is None:
            return None
        elif self.__head is self.__tail:
            self.__head = None
        else:
            self.__head = value.next
            value.next = None
        return value

    def peek(self):
        value = self.__head
        if value is None:
            return None
